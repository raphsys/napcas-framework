1. Guide d’installation du projet
Cloner le dépôt

bash
Copier
Modifier
git clone https://…/napcas-framework.git
cd napcas-framework
Installer les dépendances système

bash
Copier
Modifier
# C++ : Eigen, HDF5, stb_image, pybind11
sudo apt update
sudo apt install -y \
  build-essential cmake \
  libeigen3-dev libhdf5-dev libhdf5-cpp-dev \
  libstb-dev doxygen graphviz \
  python3-dev python3-venv python3-pip
Compiler la bibliothèque et le binding Python

bash
Copier
Modifier
# C++ core + extension Python
cd cpp
mkdir -p build && cd build
cmake .. 
make -j$(nproc)

# Copier/lier l’extension dans le package Python
cd ../../python/napcas
ln -sf ../../cpp/build/_napcas*.so _napcas.so

# Installer le package Python en editable
cd ../..
python3 -m venv venv
source venv/bin/activate
pip install --upgrade pip
pip install -e .
Vérifier l’installation

bash
Copier
Modifier
python - <<EOF
import napcas
t = napcas.Tensor([2,3])
print("OK, shape:", t.shape())
EOF
